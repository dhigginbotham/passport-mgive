    api.selectOperation = (req, res, next) ->

  mGive = new m()

  if req.body? && req.body.op?

    switch req.body.op
      
      when "login" 
      then api.getResponse mGive.login(req.body.username, req.body.password), (err, resp) ->
        next err, null if err?
        req.mGiveApi = resp if resp?
        console.log resp if resp?
        next()
      
      when "GetPollCampaigns" 
      then api.getResponse mGive.GetPollCampaigns(req.body.accessToken, req.body.sessionId), (err, resp) ->
        next err, null if err?
        req.mGiveApi = resp if resp?
        console.log resp if resp?
        next()
      
      when "GetPollQuestions" 
      then api.getResponse mGive.GetPollQuestions(req.body.accessToken, req.body.sessionId, req.body.campaignId), (err, resp) ->
        next err, null if err?
        req.mGiveApi = resp if resp?
        console.log resp if resp?
        next()
      
      when "GetQuestionResponses" 
      then api.getResponse mGive.GetQuestionResponses(req.body.type, req.body.accessToken, req.body.sessionId, req.body.campaignId), (err, resp) ->
        next err, null if err?
        req.mGiveApi = resp if resp?
        console.log resp if resp?
        next()
      
      when "GetAnswerStats" 
      then api.getResponse mGive.GetAnswerStats(req.body.questionId, req.body.accessToken, req.body.sessionId), (err, resp) ->
        next err, null if err?
        req.mGiveApi = resp if resp?
        console.log resp if resp?
        next()
      
      when "ApprovedQuestionResponses" 
      then api.getResponse mGive.ApprovedQuestionResponses(req.body.questionId, req.body.accessToken, req.body.sessionId, req.body.maxResults), (err, resp) ->
        next err, null if err?
        req.mGiveApi = resp if resp?
        console.log resp if resp?
        next()
      
      when "SetAutoApproveFlag" 
      then api.getResponse mGive.SetAutoApproveFlag(req.body.autoApprove, req.body.pollQuestionId, req.body.accessToken, req.body.sessionId), (err, resp) ->
        next err, null if err?
        req.mGiveApi = resp if resp?
        console.log resp if resp?
        next()
      
      when "GetDonationTotal" 
      then api.getResponse mGive.GetDonationTotal(req.body.username, req.body.password, req.body.campaignId, req.body.mobileNumber, req.body.startDate, req.body.endDate, req.body.accessToken), (err, resp) ->
        next err, null if err?
        req.mGiveApi = resp if resp?
        console.log resp if resp?
        next()

      else next "invalid operation attempted..", null
